\newcommand{\groupsinner}[2]{
  \IfEqCase{#1}{%
      {tri}{%
           \IfEqCase{#2}{%
           {ll}{\markRed\markBlue}}      %group (tri,ll)
           [\markRed\markWhite]          %group (tri)
      }%
  }[%
           \IfEqCase{#2}{%
           {ll}{\markWhite\markBlue}}     %group (ll)
  ]
}
